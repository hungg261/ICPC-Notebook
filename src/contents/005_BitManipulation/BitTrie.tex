% Binary Trie (Bit Trie)

Prefix tree structure for processing 32-bit integers, useful for Max XOR queries or subset elements monitoring in $O(B)$ where $B$ represents the bit-width.
\begin{lstlisting}
struct BitTrie {
    static const int BITS = 30; // 30 bits for values up to 10^9
    struct Node {
        int next[2];
        int cnt;
        Node() { next[0] = next[1] = 0; cnt = 0; }
    };
    vector<Node> trie;

    BitTrie() { trie.push_back(Node()); }

    void insert(int x) {
        int u = 0;
        trie[u].cnt++;
        for (int i = BITS; i >= 0; i--) {
            int bit = (x >> i) & 1;
            if (!trie[u].next[bit]) {
                trie[u].next[bit] = trie.size();
                trie.push_back(Node());
            }
            u = trie[u].next[bit];
            trie[u].cnt++;
        }
    }

    void remove(int x) {
        int u = 0;
        trie[u].cnt--;
        for (int i = BITS; i >= 0; i--) {
            int bit = (x >> i) & 1;
            u = trie[u].next[bit];
            trie[u].cnt--;
        }
    }

    int max_xor(int x) {
        int u = 0, res = 0;
        for (int i = BITS; i >= 0; i--) {
            int bit = (x >> i) & 1;
            int toggle = bit ^ 1;
            if (trie[u].next[toggle] && trie[trie[u].next[toggle]].cnt > 0) {
                res |= (1 << i);
                u = trie[u].next[toggle];
            } else {
                u = trie[u].next[bit];
            }
        }
        return res;
    }
};
\end{lstlisting}

Initialize via \lstinline|BitTrie trie| to insert elements. Use \lstinline|trie.max_xor(x)| to find the maximum possible value of $x \oplus y$ among all elements $y$ currently stored in the structure.
