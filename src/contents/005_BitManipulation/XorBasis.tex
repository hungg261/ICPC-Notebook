% Linear Basis (XOR Basis)

A vector space structure used to find the maximum possible XOR sum subset or check independence of integers in $O(B)$ per operation, where $B$ is the bit-width.
\begin{lstlisting}
struct XorBasis {
    static const int BITS = 60; // Up to 2^60
    long long basis[BITS];
    int sz = 0;

    XorBasis() {
        memset(basis, 0, sizeof(basis));
    }

    bool insert(long long mask) {
        for (int i = BITS - 1; i >= 0; i--) {
            if ((mask >> i) & 1) {
                if (!basis[i]) {
                    basis[i] = mask;
                    sz++;
                    return true;
                }
                mask ^= basis[i];
            }
        }
        return false; // Dependent element, not inserted
    }

    long long max_xor() {
        long long res = 0;
        for (int i = BITS - 1; i >= 0; i--) {
            if ((res ^ basis[i]) > res) {
                res ^= basis[i];
            }
        }
        return res;
    }
};
\end{lstlisting}

Initialize via \lstinline|XorBasis b|. Use \lstinline|b.insert(x)| to add an element to the vector space. Call \lstinline|b.max_xor()| to compute the maximum possible XOR sum that can be formed using any subset of the inserted elements.
