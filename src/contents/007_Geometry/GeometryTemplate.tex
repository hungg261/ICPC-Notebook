% Geometry Templates (Integer \& Double)

Complete 2D geometry templates. The integer version guarantees 100% precision without floating-point errors, while the double version handles fractional coordinates and intersections using EPS.

\begin{lstlisting}
// 1. INTEGER GEOMETRY TEMPLATE
struct PointInt {
    long long x, y;
    PointInt(long long x = 0, long long y = 0) : x(x), y(y) {}
    PointInt operator+(const PointInt& o) const { return {x + o.x, y + o.y}; }
    PointInt operator-(const PointInt& o) const { return {x - o.x, y - o.y}; }
};

long long dot(PointInt a, PointInt b) { return a.x * b.x + a.y * b.y; }
long long cross(PointInt a, PointInt b) { return a.x * b.y - a.y * b.x; }
long long dist2(PointInt a, PointInt b) { return dot(a - b, a - b); }

int ccw(PointInt a, PointInt b, PointInt c) {
    long long cr = cross(b - a, c - a);
    if (cr == 0) return 0;
    return (cr > 0) ? 1 : -1;
}

bool on_segment(PointInt p, PointInt a, PointInt b) {
    return ccw(a, b, p) == 0 && dot(a - p, b - p) <= 0;
}

bool intersect(PointInt a, PointInt b, PointInt c, PointInt d) {
    if (on_segment(c, a, b) || on_segment(d, a, b) || on_segment(a, c, d) || on_segment(b, c, d)) return true;
    return ccw(a, b, c) * ccw(a, b, d) < 0 && ccw(c, d, a) * ccw(c, d, b) < 0;
}
\end{lstlisting}

\begin{lstlisting}
// 2. DOUBLE GEOMETRY TEMPLATE
const double EPS = 1e-9;

struct PointDbl {
    double x, y;
    PointDbl(double x = 0, double y = 0) : x(x), y(y) {}
    PointDbl operator+(const PointDbl& o) const { return {x + o.x, y + o.y}; }
    PointDbl operator-(const PointDbl& o) const { return {x - o.x, y - o.y}; }
    PointDbl operator*(double c) const { return {x * c, y * c}; }
    PointDbl operator/(double c) const { return {x / c, y / c}; }
};

double dot(PointDbl a, PointDbl b) { return a.x * b.x + a.y * b.y; }
double cross(PointDbl a, PointDbl b) { return a.x * b.y - a.y * b.x; }
double dist2(PointDbl a, PointDbl b) { return dot(a - b, a - b); }
double dist(PointDbl a, PointDbl b) { return sqrt(dist2(a, b)); }

int ccw(PointDbl a, PointDbl b, PointDbl c) {
    double cr = cross(b - a, c - a);
    if (abs(cr) < EPS) return 0;
    return (cr > 0) ? 1 : -1;
}

bool on_segment(PointDbl p, PointDbl a, PointDbl b) {
    return ccw(a, b, p) == 0 && dot(a - p, b - p) <= EPS;
}

bool intersect(PointDbl a, PointDbl b, PointDbl c, PointDbl d) {
    if (on_segment(c, a, b) || on_segment(d, a, b) || on_segment(a, c, d) || on_segment(b, c, d)) return true;
    return ccw(a, b, c) * ccw(a, b, d) < 0 && ccw(c, d, a) * ccw(c, d, b) < 0;
}

PointDbl line_intersection(PointDbl a, PointDbl b, PointDbl c, PointDbl d) {
    double cp = cross(b - a, d - c);
    assert(abs(cp) > EPS);
    return a + (b - a) * (cross(c - a, d - c) / cp);
}
\end{lstlisting}
