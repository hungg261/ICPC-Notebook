% Essential Theorems

Critical theorems and properties for structural reductions and mathematical transformations.

\textbf{Graph Theory Invariants \& Reductions:}
\begin{itemize}
    \item \textbf{Dilworth:} Max antichain (no two comparable) $=$ min chains to cover poset. Reduces to Max Bipartite Matching.
    \item \textbf{Konig:} In any bipartite graph, $|Max\ Matching| = |Min\ Vertex\ Cover|$.
    \item \textbf{Independent Set:} $S$ is vertex cover iff $V \setminus S$ is independent set $\implies |Min\ Vertex\ Cover| + |Max\ Independent\ Set| = |V|$.
    \item \textbf{Gallai:} In graphs without isolated nodes, $|Max\ Matching| + |Min\ Edge\ Cover| = |V|$.
    \item \textbf{Eulerian Path:} Undirected: cycle iff all degrees even; path iff exactly 0 or 2 nodes have odd degree. Directed: cycle iff $in[u]=out[u]$ for all $u$; path iff at most one node has $out-in=1$ and at most one has $in-out=1$.
\end{itemize}

\textbf{Number Theory Identities:}
\begin{itemize}
    \item \textbf{Euler \& Fermat:} $\gcd(a, m) = 1 \implies a^{\phi(m)} \equiv 1 \pmod m$. If $p$ is prime, $a^{p-1} \equiv 1 \pmod p$.
    \item \textbf{Bezout:} $\exists x, y$ such that $ax + by = \gcd(a, b)$, solved via ExtGCD.
    \item \textbf{Wilson:} Integer $n > 1$ is prime iff $(n-1)! \equiv -1 \pmod n$.
    \item \textbf{Mobius Inversion:} If $g(n) = \sum_{d|n} f(d)$, then $f(n) = \sum_{d|n} \mu(d)g(n/d)$.
    \item \textbf{GCD Properties:} 
    \begin{itemize}
        \item $\gcd(a^n - 1, a^m - 1) = a^{\gcd(n, m)} - 1$.
        \item $\gcd(F_a, F_b) = F_{\gcd(a, b)}$
    \end{itemize}
\end{itemize}
