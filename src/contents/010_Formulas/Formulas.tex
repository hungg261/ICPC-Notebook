% Essential Mathematics Formulas

A concise compilation of counting sequences, identities, and structural counting theorems frequently encountered in competitive programming.

\subsection{Combinatorics \& Modular Precomputations}
Precompute factorials and modular inverse factorials in $O(N)$ for $O(1)$ combination queries modulo prime $M$:
$$C_n^k = \frac{n!}{k!(n-k)!} \pmod M$$
$$inv[i] = -\lfloor M/i \rfloor \cdot inv[M \pmod i] \pmod M$$

\subsection{Catalan Numbers}
Counts binary search trees with $n$ nodes, valid parenthesization pairs, and non-crossing polygon dissections.
$$C_0 = 1, \quad C_n = \sum_{i=0}^{n-1} C_i C_{n-1-i} = \frac{1}{n+1} C_{2n}^n = C_{2n}^n - C_{2n}^{n-1}$$
The first few terms: $1, 1, 2, 5, 14, 42, 132, 429, 1430, 4862$.

\subsection{Derangements (Permutations with no fixed points)}
Counts permutations of length $n$ where no element $i$ appears in its original $i$-th position.
$$D_1 = 0, \quad D_2 = 1, \quad D_n = (n-1)(D_{n-1} + D_{n-2}) = n D_{n-1} + (-1)^n$$
The first few terms: $0, 1, 2, 9, 44, 265, 1854, 14833, 133496$.

\subsection{Stirling Numbers of the Second Kind}
$S(n, k)$ represents the number of ways to partition a set of $n$ labeled elements into $k$ unlabeled non-empty subsets.
$$S(n, k) = S(n-1, k-1) + k \cdot S(n-1, k)$$
Base conditions: $S(0, 0) = 1, \quad S(n, 0) = S(0, n) = 0$.
Number of ways to partition $n$ labeled items into $k$ labeled boxes is $k! \cdot S(n, k)$.

\subsection{Lucas' Theorem}
Computes $C_n^k \pmod p$ for a small prime $p$ when $n, k \le 10^{18}$. Express $n$ and $k$ in base $p$:
$$n = n_m p^m + n_{m-1} p^{m-1} + \dots + n_0, \quad k = k_m p^m + k_{m-1} p^{m-1} + \dots + k_0$$
$$C_n^k \equiv \prod_{i=0}^{m} C_{n_i}^{k_i} \pmod p \quad (\text{where } C_{n_i}^{k_i} = 0 \text{ if } n_i < k_i)$$

\subsection{Graph Theory Counting}
\begin{itemize}
    \item \textbf{Cayley's Formula:} The number of unique labeled spanning trees in a complete graph $K_n$ is $n^{n-2}$.
    \item \textbf{Handshaking Lemma:} $\sum \deg(v) = 2 \cdot |E|$. Total number of vertices with odd degree is always even.
    \item \textbf{Euler's Planar Formula:} For any connected planar graph: $V - E + F = 2$ (where $F$ includes the outer infinite face).
\end{itemize}

\subsection{Geometry Theorems}
\begin{itemize}
    \item \textbf{Pick's Theorem:} Calculates the area $A$ of a polygon whose vertices all lie on an integer grid:
    $$A = I + \frac{B}{2} - 1$$
    where $I$ is the number of interior grid points, and $B$ is the number of boundary grid points.
    \item \textbf{Boundary Points on Segment:} The number of integer points strictly lying on the line segment connecting $(x_1, y_1)$ and $(x_2, y_2)$ including both endpoints is:
    $$\gcd(|x_1 - x_2|, |y_1 - y_2|) + 1$$
\end{itemize}
