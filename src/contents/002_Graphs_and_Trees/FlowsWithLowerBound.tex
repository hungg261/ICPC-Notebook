% Flow with Lower Bound

Graph transformation for network flows with lower bounds $[L, C]$.

\begin{enumerate}
    \item \textbf{Transform Graph:} New capacity is $C - L$. Update demands: $\text{delta}[u] -= L, \, \text{delta}[v] += L$.
    \item \textbf{Super Source ($S'$) \& Sink ($T'$):} For each vertex $u$:
    \begin{itemize}
        \item $\text{delta}[u] > 0$: Add edge $(S', u)$ with capacity $\text{delta}[u]$.
        \item $\text{delta}[u] < 0$: Add edge $(u, T')$ with capacity $-\text{delta}[u]$.
    \end{itemize}
\end{enumerate}

\textbf{Feasible Flow (Circulatory):} Run Max Flow $S' \to T'$. Feasible iff $\text{max\_flow} == \sum_{\text{delta} > 0} \text{delta}$. Real flow $= L + \text{current\_flow}$.

\textbf{Max/Min Flow with Source ($S$) and Sink ($T$):}
\begin{enumerate}
    \item Add edge $(T, S)$ with capacity $\infty$. Run Max Flow $S' \to T'$ to check feasibility.
    \item Save \lstinline|init_flow| = flow on $(T, S)$, then remove edge $(T, S)$.
    \item \textbf{Max Flow:} Run Max Flow $S \to T$. \lstinline|Total| = \lstinline|init_flow| + new\_flow($S, T$).
    \item \textbf{Min Flow:} Run Max Flow $T \to S$. \lstinline|Total| = \lstinline|init_flow| - new\_flow($T, S$).
\end{enumerate}
