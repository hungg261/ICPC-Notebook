% LCA with Binary Lifting

Lowest Common Ancestor (LCA) queries on a tree in $O(\log N)$ with $O(N \log N)$ preprocessing using binary lifting.
\begin{lstlisting}
struct LCA {
    int n, l;
    vector<int> tin, tout, depth;
    vector<vector<int>> up;
    int timer;

    LCA(int n, int root, const vector<vector<int>>& adj) : n(n), timer(0) {
        l = __lg(n) + 1;
        tin.resize(n + 1); tout.resize(n + 1); depth.resize(n + 1);
        up.assign(n + 1, vector<int>(l + 1));
        dfs(root, root, 0, adj);
    }

    void dfs(int u, int p, int d, const vector<vector<int>>& adj) {
        tin[u] = ++timer;
        depth[u] = d;
        up[u][0] = p;
        for (int i = 1; i <= l; ++i) {
            up[u][i] = up[up[u][i - 1]][i - 1];
        }
        for (int v : adj[u]) {
            if (v != p) dfs(v, u, d + 1, adj);
        }
        tout[u] = ++timer;
    }

    bool is_ancestor(int u, int v) {
        return tin[u] <= tin[v] && tout[u] >= tout[v];
    }

    int query(int u, int v) {
        if (is_ancestor(u, v)) return u;
        if (is_ancestor(v, u)) return v;
        for (int i = l; i >= 0; --i) {
            if (!is_ancestor(up[u][i], v)) u = up[u][i];
        }
        return up[u][0];
    }

    int get_dist(int u, int v) {
        return depth[u] + depth[v] - 2 * depth[query(u, v)];
    }
};
\end{lstlisting}
