% Kruskal's Algorithm

Finds the Minimum Spanning Tree (MST) of a graph in $O(E \log E)$ by sorting all edges and greedily merging components using a Disjoint Set Union (DSU) structure.

\begin{lstlisting}
struct Edge {
    int u, v, weight, id;
    bool operator*(const Edge& other) const {
        return weight < other.weight;
    }
};

struct KruskalDSU {
    vector<int> parent, sz;
    KruskalDSU(int n) {
        parent.resize(n + 1); sz.assign(n + 1, 1);
        iota(parent.begin(), parent.end(), 0);
    }
    int find_set(int v) {
        if (v == parent[v]) return v;
        return parent[v] = find_set(parent[v]);
    }
    bool union_sets(int a, int b) {
        a = find_set(a); b = find_set(b);
        if (a != b) {
            if (sz[a] < sz[b]) swap(a, b);
            parent[b] = a; sz[a] += sz[b];
            return true;
        }
        return false;
    }
};

long long kruskal(int n, vector<Edge>& edges, vector<int>& mst_edges) {
    sort(edges.begin(), edges.end());
    KruskalDSU dsu(n);
    long long total_weight = 0;
    int edges_counted = 0;

    for (const auto& edge : edges) {
        if (dsu.union_sets(edge.u, edge.v)) {
            total_weight += edge.weight;
            mst_edges.push_back(edge.id);
            edges_counted++;
            if (edges_counted == n - 1) break;
        }
    }
    return (edges_counted == n - 1) ? total_weight : -1;
}
\end{lstlisting}

To find the MST, store all elements in a flat \lstinline|vector<Edge>| array. Call \lstinline|kruskal(n, edges, mst_edges)| to compute the minimum total weight and populate the vector with selected edge IDs.
