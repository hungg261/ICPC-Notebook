% Kosaraju's Algorithm

Finds Strongly Connected Components (SCC) in a directed graph in $O(V + E)$ using two passes of Depth First Search (DFS).
\begin{lstlisting}
struct Kosaraju {
    int n, scc_count;
    vector<vector<int>> adj, adj_rev;
    vector<bool> visited;
    vector<int> order, scc_id;

    Kosaraju(int n) : n(n), scc_count(0), adj(n + 1), adj_rev(n + 1), visited(n + 1, false), scc_id(n + 1, -1) {}

    void add_edge(int u, int v) {
        adj[u].push_back(v);
        adj_rev[v].push_back(u);
    }

    void dfs1(int u) {
        visited[u] = true;
        for (int v : adj[u]) {
            if (!visited[v]) dfs1(v);
        }
        order.push_back(u);
    }

    void dfs2(int u, int cl) {
        visited[u] = true;
        scc_id[u] = cl;
        for (int v : adj_rev[u]) {
            if (!visited[v]) dfs2(v, cl);
        }
    }

    void run() {
        for (int i = 1; i <= n; i++) {
            if (!visited[i]) dfs1(i);
        }
        fill(visited.begin(), visited.end(), false);
        reverse(order.begin(), order.end());
        for (int u : order) {
            if (!visited[u]) {
                dfs2(u, ++scc_count);
            }
        }
    }
};
\end{lstlisting}

Call \lstinline|kosaraju.run()| to compute components. The total number of SCCs is stored in \lstinline|scc_count|, and the 1-based component ID for each vertex is accessible via \lstinline|scc_id[i]|.
