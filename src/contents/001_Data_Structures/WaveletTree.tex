% Wavelet Tree

A powerful data structure that supports generalized range queries (like finding the $k$-th smallest element or the number of elements in range $[l, r]$ less than or equal to $k$) in $O(\log(\max A_i))$ time.

\begin{lstlisting}
struct WaveletTree {
    int lo, hi;
    WaveletTree *l, *r;
    vector<int> b;

    // Elements must be in range [lo, hi], sequence passes by iterators
    template<typename Iter>
    WaveletTree(Iter from, Iter to, int lo, int hi) : lo(lo), hi(hi), l(nullptr), r(nullptr) {
        if (from >= to || lo == hi) return;
        int mid = (lo + hi) >> 1;
        auto f = [mid](int x) { return x <= mid; };
        b.reserve(to - from + 1);
        b.push_back(0);
        for (auto it = from; it != to; ++it) {
            b.push_back(b.back() + f(*it));
        }
        auto pivot = stable_partition(from, to, f);
        l = new WaveletTree(from, pivot, lo, mid);
        r = new WaveletTree(pivot, to, mid + 1, hi);
    }

    // Returns the k-th smallest element in range [l, r] (1-indexed k)
    int kth(int l, int r, int k) {
        if (lo == hi) return lo;
        int in_left = b[r] - b[l - 1];
        int lb = b[l - 1], rb = b[r];
        if (k <= in_left) return l->kth(lb + 1, rb, k);
        return r->kth(l - lb, r - rb, k - in_left);
    }

    // Returns count of elements in range [l, r] less than or equal to k
    int count_le(int l, int r, int k) {
        if (l > r || k < lo) return 0;
        if (hi <= k) return r - l + 1;
        int lb = b[l - 1], rb = b[r];
        return l->count_le(lb + 1, rb, k) + r->count_le(l - lb, r - rb, k);
    }

    ~WaveletTree() {
        delete l;
        delete r;
    }
};
\end{lstlisting}

Initialize via \lstinline|WaveletTree T(a.begin(), a.end(), min_val, max_val)|. Query operations like \lstinline|T.kth(l, r, k)| and \lstinline|T.count_le(l, r, k)| expect 1-indexed range boundaries.
