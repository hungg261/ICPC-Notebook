% DSU with Rollback

Disjoint Set Union with history logging to undo merge operations. Path compression is disabled to preserve tree structural history, maintaining operations in $O(\log N)$.
\begin{lstlisting}
struct DSU_Rollback {
    vector<int> parent, sz;
    vector<pair<int, int>> history;
    DSU_Rollback(int n) {
        parent.resize(n + 1); sz.assign(n + 1, 1);
        iota(parent.begin(), parent.end(), 0);
    }
    int find_set(int v) {
        while (v != parent[v]) v = parent[v];
        return v;
    }
    int get_time() { return history.size(); }
    bool union_sets(int a, int b) {
        a = find_set(a); b = find_set(b);
        if (a == b) return false;
        if (sz[a] < sz[b]) swap(a, b);
        history.push_back({b, parent[b]});
        history.push_back({~a, sz[a]});
        parent[b] = a; sz[a] += sz[b];
        return true;
    }
    void rollback(int t) {
        while (history.size() > t) {
            auto [u, val] = history.back(); history.pop_back();
            if (u < 0) sz[~u] = val;
            else parent[u] = val;
        }
    }
};
\end{lstlisting}

Call \lstinline|int t = dsu.get_time()| to save the state checkpoint before processing speculative edges. Pass \lstinline|t| to \lstinline|dsu.rollback(t)| to revert the components back to that structural checkpoint.
