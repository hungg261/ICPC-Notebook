% Implicit Treap

Indexed by array positions instead of keys. Supports range operations (e.g., reverse) and split/merge in $O(\log N)$ expected time.

\begin{lstlisting}
struct Node {
    int val, prior, sz;
    bool rev;
    Node *l, *r;
    Node(int val) : val(val), prior(rng()), sz(1), rev(false), l(nullptr), r(nullptr) {}
};

int get_sz(Node* t) { return t ? t->sz : 0; }
void upd_sz(Node* t) { if (t) t->sz = 1 + get_sz(t->l) + get_sz(t->r); }

void push(Node* t) {
    if (t && t->rev) {
        t->rev = false;
        swap(t->l, t->r);
        if (t->l) t->l->rev ^= true;
        if (t->r) t->r->rev ^= true;
    }
}

void split(Node* t, int k, Node*& l, Node*& r) {
    if (!t) { l = r = nullptr; return; }
    push(t);
    if (get_sz(t->l) < k) {
        split(t->r, k - get_sz(t->l) - 1, t->r, r);
        l = t; upd_sz(l);
    } else {
        split(t->l, k, l, t->l);
        r = t; upd_sz(r);
    }
}

void merge(Node*& t, Node* l, Node* r) {
    push(l); push(r);
    if (!l || !r) t = l ? l : r;
    else if (l->prior > r->prior) {
        merge(l->r, l->r, r);
        t = l; upd_sz(t);
    } else {
        merge(r->l, l, r->l);
        t = r; upd_sz(t);
    }
}

void reverse_range(Node*& t, int ql, int qr) {
    Node *l1, *r1, *l2, *r2;
    split(t, ql, l1, r1);
    split(r1, qr - ql + 1, l2, r2);
    if (l2) l2->rev ^= true;
    merge(r1, l2, r2);
    merge(t, l1, r1);
}
\end{lstlisting}
