% Aho-Corasick Automaton

Multi-pattern string matching automaton built on top of a Trie. It constructs suffix links and dictionary links using BFS in $O(\sum |pattern|)$ time, allowing simultaneous matching of multiple strings in $O(|text|)$.

\begin{lstlisting}
struct AhoCorasick {
    static const int ALPHABET_SIZE = 26;
    struct Node {
        int next[ALPHABET_SIZE];
        int link = -1; // Suffix link
        int exit_link = -1; // Dict link to nearest pattern end node
        vector<int> pattern_ids; // IDs of patterns ending at this node

        Node() { memset(next, -1, sizeof(next)); }
    };

    vector<Node> trie;
    AhoCorasick() { trie.push_back(Node()); }

    void insert(const string& s, int id) {
        int u = 0;
        for (char c : s) {
            int idx = c - 'a';
            if (trie[u].next[idx] == -1) {
                trie[u].next[idx] = trie.size();
                trie.push_back(Node());
            }
            u = trie[u].next[idx];
        }
        trie[u].pattern_ids.push_back(id);
    }

    void build() {
        queue<int> q;
        for (int i = 0; i < ALPHABET_SIZE; i++) {
            if (trie[0].next[i] != -1) {
                trie[trie[0].next[i]].link = 0;
                q.push(trie[0].next[i]);
            } else {
                trie[0].next[i] = 0; // Optimization: transform Trie into a DFA transition table
            }
        }

        while (!q.empty()) {
            int u = q.front(); q.pop();
            for (int i = 0; i < ALPHABET_SIZE; i++) {
                int v = trie[u].next[i];
                if (v != -1) {
                    trie[v].link = trie[trie[u].link].next[i];
                    
                    // Construct dictionary exit link
                    int slink = trie[v].link;
                    if (!trie[slink].pattern_ids.empty()) trie[v].exit_link = slink;
                    else trie[v].exit_link = trie[slink].exit_link;

                    q.push(v);
                } else {
                    trie[u].next[i] = trie[trie[u].link].next[i];
                }
            }
        }
    }
};
\end{lstlisting}

Initialize via \lstinline|AhoCorasick ac|, call \lstinline|ac.insert(s, id)| for all patterns, and then run \lstinline|ac.build()| once. When parsing a text, traverse using the modified transition matrix \lstinline|u = ac.trie[u].next[c - 'a']|. Use the \lstinline|exit_link| chain at each step to collect all matched dictionary patterns efficiently.
