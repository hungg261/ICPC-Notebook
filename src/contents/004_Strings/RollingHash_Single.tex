% String Single Hashing

Anti-hash-kill String Hashing implementation using dynamic base generation under a single secure random prime.

\begin{lstlisting}
mt19937 rng(chrono::steady_clock::now().time_since_epoch().count());

struct StringHash {
    long long MOD = 1e9 + 7;
    long long BASE;
    vector<long long> h, p;

    StringHash(const string& s) {
        BASE = (rng() % (MOD - 300) + 300) | 1;
        int n = s.size();
        h.resize(n + 1, 0);
        p.resize(n + 1, 1);
        for (int i = 0; i < n; i++) {
            h[i + 1] = (h[i] * BASE + s[i]) % MOD;
            p[i + 1] = (p[i] * BASE) % MOD;
        }
    }

    long long get(int l, int r) {
        l++; r++;
        long long res = (h[r] - h[l - 1] * p[r - l + 1]) % MOD;
        if (res < 0) res += MOD;
        return res;
    }
};
\end{lstlisting}

To verify matching substrings over range $[l, r]$, initialize via \lstinline|StringHash hasher(s)| and query using 0-indexed values \lstinline|hasher.get(l, r)|.
