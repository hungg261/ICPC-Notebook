% Suffix Automaton (SAM)

A directed acyclic word graph that represents all suffixes of a string in $O(N)$ time and space. Each state represents a set of substrings that share the same end-positions (\lstinline|endpos|).

\begin{lstlisting}
struct State {
    int len, link;
    int next[26];
};

const int MAXLEN = 100005;
State st[MAXLEN * 2];
int sz, last;

void sam_init() {
    st[0].len = 0;
    st[0].link = -1;
    memset(st[0].next, -1, sizeof(st[0].next));
    sz = 1;
    last = 0;
}

void sam_extend(char c) {
    int cur = sz++;
    st[cur].len = st[last].len + 1;
    memset(st[cur].next, -1, sizeof(st[cur].next));
    
    int p = last;
    while (p != -1 && st[p].next[c - 'a'] == -1) {
        st[p].next[c - 'a'] = cur;
        p = st[p].link;
    }
    
    if (p == -1) {
        st[cur].link = 0;
    } else {
        int q = st[p].next[c - 'a'];
        if (st[p].len + 1 == st[q].len) {
            st[cur].link = q;
        } else {
            int clone = sz++;
            st[clone].len = st[p].len + 1;
            memcpy(st[clone].next, st[q].next, sizeof(st[q].next));
            st[clone].link = st[q].link;
            while (p != -1 && st[p].next[c - 'a'] == q) {
                st[p].next[c - 'a'] = clone;
                p = st[p].link;
            }
            st[q].link = st[cur].link = clone;
        }
    }
    last = cur;
}
\end{lstlisting}

\begin{itemize}
    \item Call \lstinline|sam_init()| once before building the automaton.
    \item Call \lstinline|sam_extend(c)| sequentially for each character in the string to construct the SAM online.
    \item The total number of states in the automaton is at most $2N - 1$. Ensure \lstinline|MAXLEN| is at least the maximum string length.
    \item Useful for complex string operations such as substring counting, longest common substring of multiple strings, and finding the lexicographically $k$-th substring.
\end{itemize}
