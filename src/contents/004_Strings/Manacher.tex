% Manacher's Algorithm

Computes the longest palindromic substring centered at each position of a string in $O(N)$ time by utilizing previously calculated symmetric palindrome radius bounds.

\begin{lstlisting}
vector<int> manacher(string s) {
    string t = "#";
    for (char c : s) {
        t += c;
        t += "#";
    }
    int n = t.length();
    vector<int> p(n, 0);
    int c = 0, r = 0;
    
    for (int i = 0; i < n; i++) {
        int mirror = 2 * c - i;
        if (i < r) {
            p[i] = min(r - i, p[mirror]);
        }
        while (i + p[i] + 1 < n && i - p[i] - 1 >= 0 && t[i + p[i] + 1] == t[i - p[i] - 1]) {
            p[i]++;
        }
        if (i + p[i] > r) {
            c = i;
            r = i + p[i];
        }
    }
    return p;
}
\end{lstlisting}

Pass a \lstinline|string s| to \lstinline|manacher(s)|. The function internally pads the string with \lstinline|'#'| characters to uniformly handle both odd and even length palindromes. The actual length of the maximum palindrome centered at index \lstinline|i| in the transformed array is exactly equal to \lstinline|p[i]|.
